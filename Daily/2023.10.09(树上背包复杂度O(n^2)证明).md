```cpp
void dfs( int x , int fa )
{
    siz[x] = 1 ;
    for( auto u : g[x] )
    {
        if( u == fa ) continue ;
        dfs( u , x ) ;

        for( int j = 1 ; j <= siz[x] ; j ++ )
        {
            for( int k = 1 ; k <= siz[u] ; k ++ )
            {
                tot ++ ;
            }
        }
        siz[x] += siz[u] ;
    }

}
```
