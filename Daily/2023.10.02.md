#include<bits/stdc++.h>
using namespace std ;
#define ll long long
const ll N = 8e3+9 ;
const ll inf = 1e18 ;
const ll mod = 998244353 ;
ll a[ N ] , n ;
ll f[ N ][ N ] ;
ll q[ N ] , idx[ N ] , p1 , p2 ;
void solve(){

    cin >> n ; for( int i = 1 ; i <= n ; i ++ ) cin >> a[ i ] ;

    for( int i = 1 ; i <= n ; i ++ )
        for( int j = 1 ; j <= n ; j ++ )
            if( i != j ) f[ i ][ j ] = inf ;

    for( int r = 2 ; r <= n ; r ++ )
    {
        // 起始区间[r-1,r] , k = r-1 ， k ↓
        // f[l][r] = min( f[l][k]+a[k] , q.top()  )
        // q : f[j+1][r]+a[j] 的最小值
        // k : f[l][k] >= f[k+1][r] 的第一个 k
        int k = r-1 ; p1 = 1 ; p2 = 0 ;
        for( int l = r-1 ; l >= 1 ; l -- )
        {
            /*
            for( int k = l ; k < r ; k ++ )
                f[l][r] = min( f[l][r] , max(f[l][k],f[k+1][r]) + a[k] ) ;
            */

            //随着l的减小，决策点k将左移或不变
            // 若 k 向左移
            // f[l][k]+a[k] 单调递减，决策时选 k
            // f[k+1][r]+a[k] 无单调性，决策时选单调队列首

            while( k >= l && f[l][k] >= f[k+1][r] ) k-- ;

            while( p1 <= p2 && idx[p1] >= k+1 ) p1 ++ ;

            f[l][r] = min( f[l][k+1]+a[k+1] , ( p1 <= p2 ? q[p1] : inf ) ) ;

            while( p1 <= p2 && q[p2] >= f[l][r]+a[l-1] ) p2 -- ;
            q[++p2] = f[l][r]+a[l-1] ; idx[p2] = l-1 ;
        }
    }

    cout << f[1][n] << "\n" ;
}
int main(){
    ios::sync_with_stdio(false) ; cin.tie(0) ; cout.tie(0) ;
    ll tt = 1 ; cin >> tt ;
    while( tt-- ) solve() ;
    return 0 ;
}
